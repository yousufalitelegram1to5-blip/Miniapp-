<script>
    let currentBalance = 0;
    function startVideoAd() {
        const player = document.getElementById('videoPlayer');
        const countdownText = document.getElementById('countdown');
        player.style.display = 'flex';
        let timeLeft = 15;
        const timer = setInterval(() => {
            timeLeft--;
            countdownText.innerText = `🎬 বিজ্ঞাপন চলছে... ${timeLeft} সেকেন্ড বাকি`;
            if (timeLeft <= 0) {
                clearInterval(timer);
                player.style.display = 'none';
                currentBalance += 20;
                document.getElementById('balance').innerText = currentBalance;
                alert('অভিনন্দন! আপনি সফলভাবে ভিডিওটি দেখেছেন এবং ৳২০ পেয়েছেন।');
            }
        }, 1000);
    }
    function showReferLink() {
        alert('আপনার রেফারাল কোড: REF' + Math.floor(100000 + Math.random() * 900000) + '\n\nএটি বন্ধুদের সাথে শেয়ার করুন!');
    }
    function withdrawNotice() {
        alert('সর্বনিম্ন উত্তোলন সীমা ৳৫০০। আপনার ব্যালেন্স পর্যাপ্ত হলে বিকাশ বা রকেটে নিতে পারবেন।');
    }
</script>
